# Electricity Bill Calculator

units = int(input("Enter total electricity units consumed: "))

# Calculate bill amount
if units <= 100:
    bill = units * 5
elif units <= 200:
    bill = (100 * 5) + ((units - 100) * 7)
else:
    bill = (100 * 5) + (100 * 7) + ((units - 200) * 10)

final_bill = bill

# Apply surcharge if bill exceeds ₹2000
if bill > 2000:
    final_bill = bill + (bill * 0.05)

# Apply minimum bill condition
if final_bill < 500:
    final_bill = 500

# Display results
print("\n----- Electricity Bill -----")
print("Total Units:", units)
print("Bill Amount: ₹", bill)
print("Final Payable Amount: ₹", final_bill)

'''Output:-
Enter total electricity units consumed: 150

Total Units: 150
Bill Amount: ₹ 850
Final Payable Amount: ₹ 850
'''
