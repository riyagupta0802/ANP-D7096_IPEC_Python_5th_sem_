# List Processing

numbers = []

# Input 20 integers
for i in range(20):
    num = int(input("Enter number: "))
    numbers.append(num)

# a) Largest number
print("Largest Number:", max(numbers))

# b) Second largest number
unique = list(set(numbers))
unique.sort()
print("Second Largest Number:", unique[-2])

# c) Smallest number
print("Smallest Number:", min(numbers))

# d) Remove duplicate elements
no_duplicates = []
for num in numbers:
    if num not in no_duplicates:
        no_duplicates.append(num)
print("List without Duplicates:", no_duplicates)

# e) Print even numbers
print("Even Numbers:")
for num in numbers:
    if num % 2 == 0:
        print(num, end=" ")
print()

# f) Print numbers divisible by both 3 and 5
print("Numbers Divisible by 3 and 5:")
for num in numbers:
    if num % 3 == 0 and num % 5 == 0:
        print(num, end=" ")
print()

# g) Reverse the list without using reverse()
print("Reversed List:")
for i in range(len(numbers) - 1, -1, -1):
    print(numbers[i], end=" ")
