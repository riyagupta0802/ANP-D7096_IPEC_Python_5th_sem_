#Student Result Management
students = {}

# Taking student details from user
n = int(input("Enter number of students: "))

for i in range(n):
    roll = int(input("Enter Roll No: "))
    marks = int(input("Enter Marks: "))
    students[roll] = marks

# Find topper
topper = max(students, key=students.get)
print("\nTopper:")
print("Roll No:", topper, "Marks:", students[topper])

# Find average marks
average = sum(students.values()) / len(students)
print("\nAverage Marks:", average)

# Display students scoring above average
print("\nStudents scoring above average:")
for roll, marks in students.items():
    if marks > average:
        print("Roll No:", roll, "Marks:", marks)

# Count failed students
failed = 0
for marks in students.values():
    if marks < 35:
        failed += 1

print("\nNumber of failed students:", failed)

# Display grades
print("\nStudent Grades:")
for roll, marks in students.items():
    if marks >= 90:
        grade = "A"
    elif marks >= 75:
        grade = "B"
    elif marks >= 60:
        grade = "C"
    elif marks >= 35:
        grade = "D"
    else:
        grade = "Fail"

    print("Roll No:", roll, "Marks:", marks, "Grade:", grade)
